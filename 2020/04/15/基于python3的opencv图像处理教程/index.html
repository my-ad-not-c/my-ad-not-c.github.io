import cv2 as cv
import numpy as np

def video_demo():
    capture = cv.VideoCapture(0)
    while(True):
        ret, frame = capture.read()  # 捕捉摄像头
        frame = cv.flip(frame, 1)    # 不镜像
        cv.imshow("video", frame)
        c = cv.waitKey(50)           #
        if c == 27:
            break


def get_image_info(image):
    print(type(image))				# 图像类型
    print(image.shape) 				# 形状
    print(image.size)				# 尺寸
    print(image.dtype)				# 每个通道字节
    pixel_data = np.array(image)	 #获取像素
    print(pixel_data)   			# 像素数据

import cv2 as cv

def access_pixel(image):
    print(image.shape)
    height = image.shape[0]
    width = image.shape[1]
    channels = image.shape[2]
    print("width: %s, height: %s, channels: %s"%(height, width, channels))
    for row in range(height):
        for col in range(width):
            for c in range(channels):
                pv = image[row, col, c]
                image[row, col, c] = 255 - pv
    cv.imshow("pixel_demo", image)

def extrace_object():
    capture = cv.VideoCapture("F:\Document\Videos\英雄时刻.mp4")
    while True:
        ret, frame = capture.read()   # 两个参数分别表示： 有没有读到文件， 一帧图像
        if ret == False:
            break
        hsv = cv.cvtColor(frame, cv.COLOR_BGR2HSV)
        lower_hsv = np.array([32, 43, 46])
        upper_hsv = np.array([43, 255, 255])
        mask = cv.inRange(hsv, lowerb=lower_hsv, upperb=upper_hsv)
        cv.imshow("video", frame)
        cv.imshow("mask", mask)
        c = cv.waitKey(40)
        if c == 27:
            break


def color_space_demo(image):  # 几种色彩空间转化
    gray = cv.cvtColor(image, cv.COLOR_BGR2BGRA)
    cv.imshow("gray", gray)
    hsv = cv.cvtColor(image, cv.COLOR_BGR2HSV)
    cv.imshow("hsv", hsv)
    yuv = cv.cvtColor(image, cv.COLOR_BGR2YUV)
    cv.imshow("yuv", yuv)

# 加减乘除
def add_demo(m1, m2):
    dst = cv.add(m1, m2)
    cv.imshow("add_demo", dst)

def subtract_demo(m1, m2):
    dst = cv.subtract(m1, m2)
    cv.imshow("subtrract_demo", dst)

def divide_demo(m1, m2):
    dst = cv.divide(m1, m2)
    cv.imshow("divide_demo", dst)

def multiply_demo(m1, m2):
    dst = cv.multiply(m1, m2)
    cv.imshow("multiply_demo", dst)


# 逻辑运算
def logic_demo(m1, m2):
    dst1 = cv.bitwise_and(m1, m2)
    dst2 = cv.bitwise_or(m1, m2)
    dst3 = cv.bitwise_not(m1)

def contrast_brightness_demo(image, c, b):
    h, w, ch = image.shape
    blank = np.zeros([h, w ,ch], image.dtype)
    dst = cv.addWeighted(image, c, blank, 1-c, b)

    cv.imshow("con-bri-demo", dst)
