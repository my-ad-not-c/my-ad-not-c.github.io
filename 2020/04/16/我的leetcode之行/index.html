8_字符串转整数

# -*- coding = =utf-8 -*-
# @Time : 2020/4/7 11:14
# @Author : 宋子龙
# @File : 8_字符串转整数.py
# @Software : PyCharm
# 自动机思想
INT_MAX = 2 ** 31 - 1
INT_MIN = - 2 ** 31

class Aut0m0t0n:

    def __init__(self):
        self.state = 'start'     # 自动机状态参数
        self.sign = 1           # 自动机有无符号参数
        self.ans = 0
        self.table = {
            'start': ['start','signed','in_number','end'],
            'signed': ['end','end','in_number','end'],
            'in_number': ['end','end','in_number','end'],
            'end': ['end','end','end','end']
        }                         # 自动机转移表

    def get_col(self, c):           # 判断字符类型
        if c.isspace():            # 空格返回0
            return 0
        if c == '+' or c == '-':     # 正负号返回1
            return 1
        if c.isdigit():             # 数字返回2
            return 2
        return 3                    # 其他返回3

    def get(self, c):                  # 自动机转移函数
        self.state = self.table[self.state][self.get_col(c)]
        # 自动机转移
        if self.state == 'in_number':
            self.ans = self.ans * 10 + int(c)
            self.ans = min(self.ans, INT_MAX) if self.sign == 1 \
                else min(self.ans, -INT_MIN)
        elif self.state == 'signed':
            self.sign = 1 if c == '+' \
                else -1


class Solution:
    def myAtoi(self, str: str) -> int:
        automaton = Aut0m0t0n()
        for c in str:
            automaton.get(c)
        return automaton.sign * automaton.ans


solution = Solution()

print(solution.myAtoi('-1'))

    def movingCount(self, m: int, n: int, k: int) -> int:
        from queue import Queue
        q = Queue()
        q.put((0, 0))
        s = set()
        while not q.empty():
            x, y = q.get()
            if (x, y) not in s and 0 <= x < m and 0 <= y < n and self.digitsum(x) + self.digitsum(y) <= k:
                s.add((x, y))
                for nx, ny in [(x + 1, y), (x, y + 1)]:
                    q.put((nx, ny))
        return len(s)

    def digitsum(self,x: int):
        p = 0
        # while x >= 10:
        #     p += x % 10
        #     x = x / 10
        # p += x
        while x:
            p += x % 10
            x //= 10
        return p

class Solution:
    def generateParenthesis(self, n: int) -> List[str]:
        ans = []
        def backtrack(S, left, right):
            if len(S) == 2 * n:
                ans.append(''.join(S))
                return
            if left < n:
                S.append('(')
                backtrack(S, left+1, right)
                S.pop()
            if right < left:
                S.append(')')
                backtrack(S, left, right+1)
                S.pop()
        backtrack([], 0, 0)
        return ans
solution = Solution()
print(solution.generateParenthesis(4))

class Solution:
    def reverseWords(self, s: str) -> str:
        wordlist = s.split(sep=None, maxsplit=-1)
        # print(wordlist)
        rewordlist = reversed(wordlist)
        # print(rewordlist)
        ans = ' '.join(rewordlist)

        return ans
solution = Solution()
